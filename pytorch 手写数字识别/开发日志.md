## 定义了神经网络结构
```python
import torch


# 定义模型结构
class MNIST_CNN(torch.nn.Module):  # MNIST + Convolutional Neural Network
    def __init__(self):
        super(MNIST_CNN, self).__init__()

        # 卷积层
        self.conv1 = torch.nn.Conv2d(1, 16, kernel_size=3)
        self.conv2 = torch.nn.Conv2d(16, 32, kernel_size=3)
        self.conv3 = torch.nn.Conv2d(32, 64, kernel_size=3)

        # 池化层
        self.pool = torch.nn.MaxPool2d(2, 2)

        # 全连接层
        self.f1 = torch.nn.Linear(64 * 3 * 3, 320)
        self.f2 = torch.nn.Linear(320, 160)
        self.f3 = torch.nn.Linear(160, 40)
        self.f4 = torch.nn.Linear(40, 10)

        self.relu = torch.nn.ReLU()  # 激活函数

    def forward(self, x):  # 前向传播
        y = self.conv1(x)
        y = self.relu(y)
        y = self.pool(y)
        y = self.conv2(y)
        y = self.relu(y)
        y = self.pool(y)
        y = self.conv3(y)
        y = self.relu(y)

        y = torch.flatten(y, 1)
        
        y = self.f1(y)
        y = self.relu(y)
        y = self.f2(y)
        y = self.relu(y)
        y = self.f3(y)
        y = self.relu(y)
        y = self.f4(y)
        return y

mnist_cnn = MNIST_CNN()
```

## 加载训练数据集
```
from PIL import Image
import torch
# from torchvision import transforms  弃用
import os

# 获取文件集
train = os.listdir('train_data/train')
test = os.listdir('train_data/test')
print(len(train), len(test))

import pprint
import numpy
import tqdm
# 将数据加载进入内存


# 环境有点问题用不了torchvision所以弄了个奇妙小工具
def pil_to_tensor(pil_img):  # 来自豆包
    # 直接从PIL获取数据 → 转为tensor
    width, height = pil_img.size
    
    # 新版PyTorch无警告写法
    data = pil_img.tobytes()
    img_tensor = torch.frombuffer(data, dtype=torch.uint8)
    
    # 形状调整：单通道 [1, H, W]
    img_tensor = img_tensor.view(height, width, 1)
    img_tensor = img_tensor.permute(2, 0, 1)
    
    # 归一化 0~1
    img_tensor = img_tensor.float() / 255.0
    return img_tensor

files = [('train', f) for f in train] + [('test', f) for f in test]
files_content = {'train': [], 'test': []}

# trans = transforms.ToTensor()  弃用

for file in tqdm.tqdm(files):
    img = Image.open(os.path.join('train_data', file[0], file[1])).convert("L")
    # img = trans(img) 弃用
    img = pil_to_tensor(img)
    files_content[file[0]].append(img)

```
